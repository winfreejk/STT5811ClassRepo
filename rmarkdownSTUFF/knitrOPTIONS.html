> sims <- 10000
> size <- 25
> set.seed(123)
> M1 <- matrix(rnorm(sims*size, 10, 2), nrow = sims, byrow = TRUE)
> dim(M1)
[1] 10000    25
> xbar <- apply(M1, 1, mean)
> md <- apply(M1, 1, median)
> mean(xbar)
[1] 10.00586
> sd(xbar)
[1] 0.3931081
> median(xbar)
[1] 10.00682
> sd(md)
[1] 0.4962346
> plot(density(md, bw = 0.15), ylim = c(0, 1))
> lines(density(xbar, bw = 0.15), col = "red")

The same thing using a loop now.

> xbar2 <- numeric(sims)
> md2 <- numeric(sims)
> set.seed(123)
> for(i in 1:sims){
+   sample <- rnorm(size, 10, 2)
+   xbar2[i] <- mean(sample)
+   md2[i] <- median(sample)
+ }

[1] 10.00586
